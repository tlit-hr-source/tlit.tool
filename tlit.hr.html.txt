// ─── DATA STORE ───────────────────────────────────────────────────
const KEYS = { EMP: 'hr_employees', ATT: 'hr_attendance', LEAVE: 'hr_leaves', ACT: 'hr_activity' };

function load(key) { try { return JSON.parse(localStorage.getItem(key)) || []; } catch(e) { return []; } }
function save(key, data) { localStorage.setItem(key, JSON.stringify(data)); }

let employees = load(KEYS.EMP);
let attendance = load(KEYS.ATT);
let leaves = load(KEYS.LEAVE);
let activity = load(KEYS.ACT);

// ─── HELPERS ──────────────────────────────────────────────────────
function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2,6); }
function today() { return new Date().toISOString().split('T')[0]; }
function fmtDate(d) { if (!d) return '—'; const dt = new Date(d + 'T00:00:00'); return dt.toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric' }); }
function fmtMoney(n) { return '₹' + Number(n).toLocaleString('en-IN'); }
function initials(name) { return name.split(' ').map(w=>w[0]).join('').toUpperCase().slice(0,2); }

const COLORS = ['#eef2ff:#3b5bdb','#ebfbee:#2f9e44','#fff9db:#e67700','#f3f0ff:#6741d9','#fff5f5:#c92a2a','#e7f5ff:#1c7ed6','#fff0f6:#c2255c','#f8f0fc:#9c36b5'];
function empColor(id) { const i = employees.findIndex(e=>e.id===id); const c = COLORS[i % COLORS.length].split(':'); return { bg: c[0], text: c[1] }; }

function logActivity(msg, icon='ti-check') { activity.unshift({ msg, icon, time: new Date().toLocaleTimeString('en-IN', {hour:'2-digit',minute:'2-digit'}) }); if (activity.length > 50) activity.pop(); save(KEYS.ACT, activity); }

function toast(msg, type='success') {
  const t = document.createElement('div'); t.className = 'toast';
  const colors = { success: '#2f9e44', error: '#c92a2a', info: '#3b5bdb' };
  t.style.background = colors[type] || colors.success;
  t.innerHTML = `<i class="ti ti-${type==='error'?'alert-circle':'check'}"></i> ${msg}`;
  document.body.appendChild(t); setTimeout(() => t.remove(), 3500);
}

// ─── NAV ──────────────────────────────────────────────────────────
function navigate(page) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  document.getElementById('page-' + page).classList.add('active');
  document.querySelectorAll('.nav-item').forEach(n => { if (n.getAttribute('onclick') && n.getAttribute('onclick').includes("'"+page+"'")) n.classList.add('active'); });
  if (page === 'dashboard') renderDashboard();
  if (page === 'employees') renderEmployees();
  if (page === 'attendance') populateAttSelects();
  if (page === 'leaves') renderLeaves();
  if (page === 'payroll') renderPayroll();
  if (page === 'reports') renderReports();
}

// ─── MODAL ────────────────────────────────────────────────────────
function openModal(id) { document.getElementById(id).classList.add('open'); }
function closeModal(id) { document.getElementById(id).classList.remove('open'); }
document.querySelectorAll('.modal-overlay').forEach(m => m.addEventListener('click', e => { if (e.target === m) m.classList.remove('open'); }));

// ─── TAB SWITCH ───────────────────────────────────────────────────
function switchTab(group, tab, el) {
  document.querySelectorAll(`#page-${group} .tab`).forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  if (group === 'leave') {
    ['pending','approved','rejected','balance'].forEach(t => { const el2 = document.getElementById('leave-'+t); if(el2) el2.style.display = t===tab ? 'block' : 'none'; });
    if (tab === 'balance') populateLeaveBalSel();
  }
}

// ─── EMPLOYEES ────────────────────────────────────────────────────
function saveEmployee() {
  const id = document.getElementById('edit-emp-id').value;
  const first = document.getElementById('f-first').value.trim();
  const last  = document.getElementById('f-last').value.trim();
  const email = document.getElementById('f-email').value.trim();
  const dept  = document.getElementById('f-dept').value.trim();
  const role  = document.getElementById('f-role').value.trim();
  const doj   = document.getElementById('f-doj').value;
  const salary = parseFloat(document.getElementById('f-salary').value);

  if (!first || !last || !email || !dept || !role || !doj || isNaN(salary)) { toast('Please fill all required fields.', 'error'); return; }

  const emp = {
    id: id || uid(),
    name: first + ' ' + last,
    email, phone: document.getElementById('f-phone').value.trim(),
    dept, role, doj, salary,
    status: document.getElementById('f-status').value,
    dob: document.getElementById('f-dob').value,
    gender: document.getElementById('f-gender').value,
    address: document.getElementById('f-address').value.trim(),
    emergency: document.getElementById('f-emergency').value.trim(),
  };

  if (id) {
    const idx = employees.findIndex(e => e.id === id);
    employees[idx] = emp;
    logActivity(`Updated profile: ${emp.name}`, 'ti-edit');
    toast('Employee profile updated!');
  } else {
    emp.empId = 'EMP-' + String(employees.length + 1001);
    employees.push(emp);
    logActivity(`Added new employee: ${emp.name}`, 'ti-user-plus');
    toast(`${emp.name} added successfully!`);
  }

  save(KEYS.EMP, employees);
  closeModal('modal-add-emp');
  renderEmployees();
  renderDashboard();
  updateDeptDatalist();
}

function renderEmployees() {
  const search = (document.getElementById('emp-search')?.value || '').toLowerCase();
  const deptF = document.getElementById('emp-dept-filter')?.value || '';
  const statusF = document.getElementById('emp-status-filter')?.value || '';

  let list = employees.filter(e => {
    const match = !search || e.name.toLowerCase().includes(search) || (e.empId||'').toLowerCase().includes(search) || e.dept.toLowerCase().includes(search) || e.role.toLowerCase().includes(search) || e.email.toLowerCase().includes(search);
    const deptMatch = !deptF || e.dept === deptF;
    const statusMatch = !statusF || e.status === statusF;
    return match && deptMatch && statusMatch;
  });

  const tbody = document.getElementById('emp-tbody');
  const empty = document.getElementById('emp-empty');
  document.getElementById('emp-count-sub').textContent = `${employees.length} team members`;

  if (!list.length) { tbody.innerHTML = ''; if(empty) empty.style.display = 'block'; return; }
  if(empty) empty.style.display = 'none';

  tbody.innerHTML = list.map(e => {
    const c = empColor(e.id);
    const badge = e.status === 'Active' ? 'badge-success' : e.status === 'Probation' ? 'badge-accent' : 'badge-danger';
    return `<tr>
      <td><div class="emp-cell"><div class="avatar" style="background:${c.bg};color:${c.text}">${initials(e.name)}</div><div><div class="emp-name">${e.name}</div><div class="emp-sub">${e.email}</div></div></div></td>
      <td style="color:var(--text3)">${e.empId||'—'}</td>
      <td>${e.dept}</td>
      <td style="color:var(--text2)">${e.role}</td>
      <td style="color:var(--text3)">${fmtDate(e.doj)}</td>
      <td>${fmtMoney(e.salary)}</td>
      <td><span class="badge ${badge}">${e.status}</span></td>
      <td style="display:flex;gap:6px">
        <button class="btn btn-sm" onclick="viewProfile('${e.id}')"><i class="ti ti-eye"></i></button>
        <button class="btn btn-sm" onclick="editEmployee('${e.id}')"><i class="ti ti-edit"></i></button>
      </td>
    </tr>`;
  }).join('');

  // Populate dept filter
  const depts = [...new Set(employees.map(e => e.dept))].sort();
  const deptSel = document.getElementById('emp-dept-filter');
  if (deptSel) { const cur = deptSel.value; deptSel.innerHTML = '<option value="">All departments</option>' + depts.map(d => `<option value="${d}"${d===cur?' selected':''}>${d}</option>`).join(''); }
}

  const onLeave = attRec.filter(a => a.status === 'Leave').length;

function buildAttLog(empId, year, month) {
  const attMap = {};
  attendance.filter(a => a.empId === empId).forEach(a => { attMap[a.date] = a; });
  const days = new Date(year, month+1, 0).getDate();
  const days_arr = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
  let rows = '';
  for (let d = days; d >= 1; d--) {
    const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    const dow = new Date(dateStr + 'T00:00:00').getDay();
    const rec = attMap[dateStr];
    if (!rec && (dow===0||dow===6)) continue;
    const status = rec ? rec.status : (dow===0||dow===6 ? 'Weekend' : '—');
    const badge = status==='Present'?'badge-success':status==='Absent'?'badge-danger':status==='Leave'?'badge-warning':status==='Holiday'?'badge-purple':'badge-gray';
    const hours = (rec?.checkIn && rec?.checkOut) ? calcHours(rec.checkIn, rec.checkOut) : '—';
    rows += `<tr><td>${fmtDate(dateStr)}</td><td style="color:var(--text3)">${days_arr[dow]}</td>
      <td><span class="badge ${badge}">${status}</span></td>
      <td>${rec?.checkIn||'—'}</td><td>${rec?.checkOut||'—'}</td><td>${hours}</td>
      <td style="color:var(--text3);font-size:12px">${rec?.note||''}</td>
      <td>${rec?`<button class="btn btn-sm" onclick="deleteAttRecord('${empId}','${dateStr}')"><i class="ti ti-trash" style="font-size:12px"></i></button>`:''}</td></tr>`;
  }
  return rows || '<tr><td colspan="8" style="text-align:center;color:var(--text3);padding:20px">No records for this month</td></tr>';
}

function calcHours(ci, co) {
  const [h1,m1] = ci.split(':').map(Number); const [h2,m2] = co.split(':').map(Number);
  const diff = (h2*60+m2) - (h1*60+m1);
  return diff > 0 ? `${Math.floor(diff/60)}h ${diff%60}m` : '—';
}

function saveLeave() {

function renderLeaves() {

    const tbody = document.getElementById(`leave-${s}-tbody`);
    const empty = document.getElementById(`leave-${s}-empty`);
    if (!tbody) return;
    if (!list.length) { tbody.innerHTML = ''; if(empty) empty.style.display='block'; return; }
    if(empty) empty.style.display='none';
    tbody.innerHTML = list.map(l => {
      const emp = employees.find(e => e.id === l.empId);
      const c = emp ? empColor(l.empId) : {bg:'#eee',text:'#333'};
      const typeColor = {Casual:'badge-accent',Earned:'badge-success',Sick:'badge-warning',Maternity:'badge-purple',Paternity:'badge-purple',Compensatory:'badge-gray',Unpaid:'badge-danger'}[l.type]||'badge-gray';

        <button class="btn btn-danger" onclick="updateLeave('${l.id}','Rejected')"><i class="ti ti-x"></i> Reject</button>

  toast(`Leave ${status.toLowerCase()}!`);
  renderLeaves(); renderDashboard();
}

function renderLeaveBalance() {
  const empId = document.getElementById('leave-bal-emp')?.value;
  const body = document.getElementById('leave-bal-body');
  if (!empId) { body.innerHTML = '<div class="empty-state" style="padding:20px"><i class="ti ti-user"></i><p>Select an employee.</p></div>'; return; }

  const limits = { Casual: 12, Earned: 18, Sick: 12, Maternity: 180, Paternity: 15, Compensatory: 10, Unpaid: 0 };
  const year = new Date().getFullYear();
  const used = {};
  leaves.filter(l => l.empId === empId && l.status === 'Approved' && new Date(l.from).getFullYear() === year).forEach(l => { used[l.type] = (used[l.type]||0) + l.days; });

  body.innerHTML = Object.entries(limits).map(([type, total]) => {
    const u = used[type] || 0;
    const rem = Math.max(0, total - u);
    const pct = total ? Math.round(u/total*100) : 0;
    const color = pct>80?'var(--danger)':pct>50?'var(--warning)':'var(--success)';
    return `<div style="margin-bottom:14px">

        <span style="font-weight:500">${type} leave</span>
        <span style="color:var(--text3)">${u} used / ${total} total — <strong style="color:${color}">${rem} remaining</strong></span>
      </div>
      <div class="progress"><div class="progress-fill" style="width:${pct}%;background:${color}"></div></div>
    </div>`;
  }).join('');
}

function renderPayroll() {
  initPayrollSelects();
  const month = parseInt(document.getElementById('payroll-month-sel')?.value ?? new Date().getMonth());
  const year  = parseInt(document.getElementById('payroll-year-sel')?.value  ?? new Date().getFullYear());
  const search = (document.getElementById('payroll-search')?.value || '').toLowerCase();
  const monthName = new Date(year, month, 1).toLocaleString('en-IN', { month: 'long' });
  document.getElementById('payroll-period-sub').textContent = `${monthName} ${year} payroll`;

  // working days in month
  let workDays = 0;
  for (let d = 1; d <= new Date(year,month+1,0).getDate(); d++) {
    const dow = new Date(year, month, d).getDay();
    if (dow !== 0 && dow !== 6) workDays++;
  }

  let list = employees.filter(e => e.status !== 'Inactive' && (!search || e.name.toLowerCase().includes(search) || e.dept.toLowerCase().includes(search)));

  let totalGross = 0, totalDeduct = 0, totalNet = 0;
  const tbody = document.getElementById('payroll-tbody');
  const empty = document.getElementById('payroll-empty');

  if (!list.length) { tbody.innerHTML = ''; if(empty) empty.style.display='block'; return; }
  if(empty) empty.style.display='none';

  tbody.innerHTML = list.map(e => {
    const basic = Math.round(e.salary * 0.6);
    const hra   = Math.round(e.salary * 0.2);
    const allow = Math.round(e.salary * 0.2);
    const gross = basic + hra + allow;
    const pf    = Math.round(basic * 0.12);
    const tds   = Math.round(gross * 0.05);
    const net   = gross - pf - tds;
    totalGross += gross; totalDeduct += pf+tds; totalNet += net;
    const c = empColor(e.id);
    return `<tr>
      <td><div class="emp-cell"><div class="avatar" style="background:${c.bg};color:${c.text}">${initials(e.name)}</div><div class="emp-name">${e.name}</div></div></td>
      <td style="color:var(--text2)">${e.dept}</td>
      <td>${basic.toLocaleString('en-IN')}</td>
      <td>${hra.toLocaleString('en-IN')}</td>
      <td>${allow.toLocaleString('en-IN')}</td>
      <td><strong>${gross.toLocaleString('en-IN')}</strong></td>
      <td style="color:var(--danger-text)">−${pf.toLocaleString('en-IN')}</td>
      <td style="color:var(--danger-text)">−${tds.toLocaleString('en-IN')}</td>
      <td><strong style="color:var(--success)">${net.toLocaleString('en-IN')}</strong></td>
      <td><button class="btn btn-sm" onclick="showPayslip('${e.id}',${month},${year})"><i class="ti ti-file-text"></i> View</button></td>
    </tr>`;
  }).join('');

  document.getElementById('pr-gross').textContent = fmtMoney(totalGross);

  document.getElementById('kpi-total').textContent = employees.length;

  const onLeave = todayAtt.filter(a => a.status==='Leave').length;

// ─── CSV EXPORT ───────────────────────────────────────────────────
function exportCSV(type) {
  let csv = '', name = '';
  if (type === 'employees') {
    name = 'employees.csv';
    csv = 'Employee ID,Name,Email,Phone,Department,Role,Date of Joining,Salary,Status,Gender,Date of Birth,Address\n';
    csv += employees.map(e => [e.empId,e.name,e.email,e.phone,e.dept,e.role,e.doj,e.salary,e.status,e.gender,e.dob,`"${e.address||''}"`].join(',')).join('\n');
  } else if (type === 'attendance') {
    name = 'attendance.csv';
    csv = 'Employee ID,Employee Name,Date,Status,Check In,Check Out,Note\n';
    csv += attendance.map(a => {
      const emp = employees.find(e=>e.id===a.empId);
      return [emp?.empId||a.empId, emp?.name||'', a.date, a.status, a.checkIn||'', a.checkOut||'', a.note||''].join(',');
    }).join('\n');
  } else if (type === 'leaves') {
    name = 'leaves.csv';
    csv = 'Employee,Leave Type,From,To,Days,Reason,Applied On,Status\n';
    csv += leaves.map(l => {
      const emp = employees.find(e=>e.id===l.empId);
      return [emp?.name||'', l.type, l.from, l.to, l.days, `"${l.reason||''}"`, l.applied, l.status].join(',');
    }).join('\n');
  } else if (type === 'payroll') {
    name = 'payroll.csv';
    csv = 'Employee ID,Name,Department,Basic,HRA,Allowance,Gross,PF,TDS,Net Pay\n';
    csv += employees.filter(e=>e.status!=='Inactive').map(e => {
      const basic=Math.round(e.salary*.6), hra=Math.round(e.salary*.2), allow=Math.round(e.salary*.2);
      const gross=basic+hra+allow, pf=Math.round(basic*.12), tds=Math.round(gross*.05), net=gross-pf-tds;
      return [e.empId,e.name,e.dept,basic,hra,allow,gross,pf,tds,net].join(',');
    }).join('\n');
  }
  if (!csv) return;
  const blob = new Blob([csv], { type: 'text/csv' });
  const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = name; a.click();
  toast('CSV exported!');
}

function exportAllCSV() {
  ['employees','attendance','leaves','payroll'].forEach((t,i) => setTimeout(() => exportCSV(t), i*300));
}

  const todayStr = today();

  const laDate = document.getElementById('leave-applied'); if(laDate) laDate.value = todayStr;

    });
  }
  document.getElementById('modal-apply-leave').addEventListener('click', () => {});

  // watch for modal opens to refresh selects
  document.querySelectorAll('[onclick*="openModal"]').forEach(btn => {
    btn.addEventListener('click', () => {
      setTimeout(() => {
        populateLeaveEmpSel();

          document.getElementById('f-doj').value = todayStr;
          document.getElementById('f-status').value = 'Active';
          document.getElementById('f-gender').value = '';
        }
      }, 50);
    });
  });

  initPayrollSelects();
}
